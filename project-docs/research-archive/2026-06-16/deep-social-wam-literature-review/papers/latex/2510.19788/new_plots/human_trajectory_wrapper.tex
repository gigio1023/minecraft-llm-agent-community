\documentclass[tikz, border=0.5cm]{standalone}
\usepackage{tikz}
\usepackage{amsmath}
\usetikzlibrary{positioning, arrows.meta}

\newcommand{\TrajectoryStep}[2]{%
  \begin{tikzpicture}[baseline=(grid.base),
    node distance=0.6cm,
    arrow/.style={-Stealth, thick, gray!80},
    label/.style={midway, font=\scriptsize, fill=white, inner sep=1pt}
  ]
    \node (grid) {#1};
    \node[below=0.4cm of grid] (caption) {\scriptsize #2};
  \end{tikzpicture}%
}

\NewDocumentCommand{\TrajectorySequence}{m}{%
  \begin{tikzpicture}[node distance=1.8cm, outer sep=0pt, inner sep=0pt]
    \newcount\StepIndex
    \StepIndex=0
    \foreach \content/\label in {#1} {
      \ifnum\StepIndex=0
        \node (step\the\StepIndex) {\TrajectoryStep{\content}{\label}};
      \else
        \node (step\the\StepIndex) [right=of step\the\numexpr\StepIndex-1\relax] {\TrajectoryStep{\content}{\label}};
        \draw[arrow] (step\the\numexpr\StepIndex-1\relax.east) -- node[label] {\label} (step\the\StepIndex.west);
      \fi
      \advance\StepIndex by 1
    }
  \end{tikzpicture}%
}
