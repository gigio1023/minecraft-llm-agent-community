export const maxRequests = Math.max(...runs.map((run) => run.requests));

export function compactNumber(value) {
  if (value >= 1000000) return `${(value / 1000000).toFixed(2)}M`;
  if (value >= 1000) return `${Math.round(value / 1000)}k`;
  return `${value}`;
}

export function OutcomeStack({run}) {
  const total = run.observed + run.blocked + run.noProgress;
  const segment = (value) => `${(value / total) * 100}%`;
  const percent = (value) => `${Math.round((value / total) * 100)}%`;
  return (
    <div className="outcomeBlock" aria-label={`${run.label} outcome distribution`}>
      <div className="outcomeStack" aria-hidden="true">
        <span className="segment observed" style={{width: segment(run.observed)}} />
        <span className="segment blocked" style={{width: segment(run.blocked)}} />
        <span className="segment noProgress" style={{width: segment(run.noProgress)}} />
      </div>
      <dl className="outcomeBreakdown">
        <div className="observed">
          <dt>Observed progress</dt>
          <dd>{run.observed}/30 <span>{percent(run.observed)}</span></dd>
        </div>
        <div className="blocked">
          <dt>Blocked</dt>
          <dd>{run.blocked}/30 <span>{percent(run.blocked)}</span></dd>
        </div>
        <div className="noProgress">
          <dt>No progress</dt>
          <dd>{run.noProgress}/30 <span>{percent(run.noProgress)}</span></dd>
        </div>
      </dl>
    </div>
  );
}

export function BarRow({run, value, max, suffix = ''}) {
  return (
    <div className="barRow">
      <span className="barLabel">{run.label}</span>
      <span className="barTrack" aria-hidden="true">
        <span className={`barFill ${run.id}`} style={{width: `${(value / max) * 100}%`}} />
      </span>
      <span className="barValue">{compactNumber(value)}{suffix}</span>
    </div>
  );
}

## Experiment Setup

The run used `natural-village-spawn-v1` with seed `4167799982467607063`. Each model
received a fresh world lane, 30 cycles, the report visual profile, and Minecraft
server version `1.21.4` for `prismarine-viewer` screenshots.

The actor-facing objective was intentionally practical:

```text
From a fresh natural village-adjacent start, establish a useful work point for
continued material work. Collect wood, craft basic materials, make or use a
crafting table, recover from blockers, and leave the world in a state that can
be reviewed from inventory, screenshots, and runtime records.
```

This is why the run looks like basic Minecraft work instead of an explicit
social-simulation demo. The aim was to check whether the environment can support
later research questions, not to declare the research target solved.

## What The Model Lanes Showed

The two ModelScope lanes were useful in the part of this experiment that matters most
to the current runtime: grounded material continuity. They did not merely describe a
plan. The records show logs becoming planks and sticks, table progress appearing,
tools showing up in inventory, and the Qwen Max lane reaching cobblestone.

That does not make this a general statement about Minecraft competence or about a
provider family. It does make the ModelScope/Qwen lanes useful examples for more
low-cost natural-village runs where the project needs material behavior before
moving toward harder action-consequence and social-response questions.

## What Still Failed

The failures are the useful part of this environment check. They show where the
runtime and action surface still need work:

- movement recovery can dominate after an early crafting success;
- table-crafting and recipe handling still create avoidable friction;
- generated action-skill candidates can be rejected by schema checks before
  a Minecraft action executes;
- visual review needs camera caveats beside screenshots, not hidden in a separate log;
- the objective was too broad for a clean short-run causal story.

For the next version of this experiment, I would make the target sharper:

```text
Place a crafting table and maintain a village-adjacent work point with bounded
material follow-through.
```

That would still produce useful material records without mixing workbench, shelter,
storage, navigation recovery, and long-horizon continuity into one run.
