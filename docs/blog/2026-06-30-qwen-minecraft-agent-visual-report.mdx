export function compactNumber(value) {
  if (value >= 1000000) return `${(value / 1000000).toFixed(2)}M`;
  if (value >= 1000) return `${Math.round(value / 1000)}k`;
  return `${value}`;
}

export function OutcomeStack({run}) {
  const total = run.verified + run.blocked + run.noProgress;
  const segment = (value) => `${(value / total) * 100}%`;
  return (
    <div className="outcomeStack" aria-label={`${run.label} outcome distribution`}>
      <span className="segment verified" style={{width: segment(run.verified)}} title={`${run.verified} verified progress`} />
      <span className="segment blocked" style={{width: segment(run.blocked)}} title={`${run.blocked} blocked`} />
      <span className="segment noProgress" style={{width: segment(run.noProgress)}} title={`${run.noProgress} no progress`} />
    </div>
  );
}

  return (
    <div className="barRow">

      <span className="barValue">{compactNumber(value)}{suffix}</span>
    </div>
  );
}

## What The Model Lanes Showed

The two ModelScope lanes were useful in the part of this experiment that matters most
to the current runtime: grounded material continuity. They did not merely describe a
plan. They left checkable traces: logs became planks and sticks, table evidence
appeared, tools showed up in inventory, and the Qwen Max lane reached cobblestone.

That does not make this a general statement about Minecraft competence or about a
provider family. It does make the ModelScope/Qwen lanes useful examples for more
low-cost evidence-producing runs where the project needs artifact-heavy material
behavior before moving toward harder action-consequence and social-response questions.

## What Still Failed

The failures are part of the value of the experiment. They show where the runtime and
action surface still need work:

- movement recovery can dominate after an early crafting success;
- table-crafting and recipe contracts still create avoidable friction;
- generated action-skill candidates can be rejected by schema/contract checks before
  a Minecraft action executes;
- visual review needs camera caveats beside screenshots, not hidden in a separate log;
- the objective was too broad for a clean 30-cycle causal story.

For the next version of this experiment, I would make the target sharper:

```text
Place a crafting table and maintain a village-adjacent work point with bounded
material follow-through.
```

That would still produce useful material evidence without mixing workbench, shelter,
storage, navigation recovery, and long-horizon continuity into one 30-cycle lane.
